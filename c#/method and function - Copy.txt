using System;

class Calculator
{
    // Method to add two numbers
    public int Add(int a, int b)
    {
        return a + b;
    }

    // Method to subtract two numbers(this is method work)
    public int Subtract(int a, int b)
    {
        return a - b;
    }
}

class Program
{
    static void Main()
    {
        Calculator calc = new Calculator();

        Console.WriteLine("Addition: " + calc.Add(5, 3));      
        Console.WriteLine("Subtraction: " + calc.Subtract(5, 3));
    }
}

using System;

class Program
{
    
    public static int Multiply(int a, int b)
    {
        return a * b;
    }

    
    public static double Divide(double a, double b)
    {
        if (b == 0)
        {
            Console.WriteLine("Cannot divide by zero.");
            return 0;
        }
        return a / b;
    }

    static void Main()
    {
        Console.WriteLine("Multiplication: " + Multiply(4, 5));   
        Console.WriteLine("Division: " + Divide(10, 2));         
    }
}
